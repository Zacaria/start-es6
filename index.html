es6 how to use examples

Es6 supported by this browser 

                            
(function supportsES6() {
    try {
        new Function("(a = 0) => a");
        document.querySelector('.browser').classList.add('is-success')
    }
    catch (err) {
        document.querySelector('.browser').classList.add('is-danger')
    }
})();

Es6 compiled with babel + gulp + browserify

Imported value from other.js : 
                            
// index.js
import {someValue} from './other';

const es6Func = (param = '') => {
  document.querySelector('.gulp').classList.add('is-success');
  document.querySelector('.gulp .imported').innerText += ' ' + param;
};

es6Func(someValue);

//other.js
export const someValue = 'foobar';
